### 4.1 exposure_rate_estimator.py

```python
#!/usr/bin/env python3
"""
事故暴露率估算模块
计算特定场景的事故率（按小时/公里）
与GB标准残余风险准则对比
"""
from dataclasses import dataclass
from typing import Dict, List, Optional
from collections import defaultdict


@dataclass
class ExposureRate:
    """暴露率计算结果"""
    operator: str
    scenario_code: str
    incident_count: int
    operating_hours: float
    collision_rate_per_hour: float
    minor_injury_rate_per_hour: float
    serious_injury_rate_per_hour: float
    fatal_rate_per_hour: float
    gb_compliance: Dict[str, bool]  # 各项GB指标合规性


class ExposureRateEstimator:
    """事故暴露率估算器"""

    # GB标准残余风险接受准则
    GB_THRESHOLDS = {
        'collision': 1e-4,       # 碰撞事故率 < 10⁻⁴/h
        'minor_injury': 1e-5,    # 轻伤事故率 < 10⁻⁵/h
        'serious_injury': 1e-6,  # 重伤事故率 < 10⁻⁶/h
        'fatal': 1e-7,           # 致命事故率 < 10⁻⁷/h
    }

    def estimate(self, incidents: List, operator: str,
                 operating_hours: float) -> ExposureRate:
        """
        估算某运营商的事故率

        Args:
            incidents: 该运营商的所有事件向量
            operator: 运营商名称
            operating_hours: 该运营商的总运营小时数

        Returns:
            暴露率结果
        """
        # 筛选该运营商的事件
        op_incidents = [i for i in incidents if i.operator == operator]

        # 按伤亡程度分类统计
        collision_count = 0
        minor_injury_count = 0
        serious_injury_count = 0
        fatal_count = 0

        for inc in op_incidents:
            if inc.fatalities > 0:
                fatal_count += 1
            elif inc.injuries > 0:
                # 简化：按严重度区分轻伤/重伤
                if inc.severity_level >= 3:  # S3+ = 重伤
                    serious_injury_count += 1
                else:
                    minor_injury_count += 1
            else:
                # 仅有物质损失或交通中断，算碰撞事件
                if inc.scenario_code.startswith(('C', 'E1')):
                    collision_count += 1

        # 计算各项事故率（按小时）
        rates = {
            'collision': collision_count / operating_hours,
            'minor_injury': minor_injury_count / operating_hours,
            'serious_injury': serious_injury_count / operating_hours,
            'fatal': fatal_count / operating_hours,
        }

        # 判定合规性
        compliance = {
            key: rates[key] < self.GB_THRESHOLDS[key]
            for key in self.GB_THRESHOLDS
        }

        # 按场景聚合
        scenario_counts = defaultdict(int)
        for inc in op_incidents:
            scenario_counts[inc.scenario_code] += 1

        return ExposureRate(
            operator=operator,
            scenario_code='ALL',  # 整体
            incident_count=len(op_incidents),
            operating_hours=operating_hours,
            collision_rate_per_hour=rates['collision'],
            minor_injury_rate_per_hour=rates['minor_injury'],
            serious_injury_rate_per_hour=rates['serious_injury'],
            fatal_rate_per_hour=rates['fatal'],
            gb_compliance=compliance
        )

    def compare_to_gb_threshold(self, rate: float, category: str) -> Dict:
        """
        与GB标准阈值对比

        Returns:
            {'threshold', 'actual', 'ratio', 'compliant', 'margin'}
        """
        threshold = self.GB_THRESHOLDS[category]
        ratio = rate / threshold
        return {
            'threshold': threshold,
            'actual': rate,
            'ratio': ratio,
            'compliant': rate < threshold,
            'margin': 1 - ratio if rate < threshold else ratio - 1
        }


if __name__ == '__main__':
    estimator = ExposureRateEstimator()
    # 示例：假设某运营商
    print("GB Standard Thresholds:")
    for category, threshold in estimator.GB_THRESHOLDS.items():
        print(f"  {category}: < {threshold:.0e}/h")
```

### 5.1 risk_pricing_model.py

```python
#!/usr/bin/env python3
"""
风险定价模型
融合PI等级、运营商评级、城市环境因子计算最终保费
"""
from dataclasses import dataclass
from typing import Dict, List, Optional
from collections import Counter


@dataclass
class PricingResult:
    """定价结果"""
    operator: str
    base_premium_per_vehicle_monthly: float
    pi_coefficient: float
    operator_coefficient: float
    city_coefficient: float
    final_premium: float
    operator_rating: str  # 1-5星
    breakdown: Dict[str, float]


class RiskPricingModel:
    """风险定价模型"""

    # 基础保费（每车每月，人民币）
    BASE_PREMIUM = 3000.0

    # PI等级保费系数
    PI_COEFFICIENTS = {
        'PI-A': 1.0,   # 不可预防，基础保费
        'PI-B': 1.5,   # 边界可预防，加成50%
        'PI-C': 3.0,   # 应可预防，加成200%
    }

    # 城市环境系数
    CITY_COEFFICIENTS = {
        # 一线复杂城市
        '北京': 1.2, '上海': 1.2, '深圳': 1.2, '广州': 1.2,
        'Beijing': 1.2, 'Shanghai': 1.2, 'Shenzhen': 1.2, 'Guangzhou': 1.2,
        # 二线城市
        '武汉': 1.0, '成都': 1.0, '杭州': 1.0, '长春': 1.0,
        'Wuhan': 1.0, 'Chengdu': 1.0,
        # 美国城市
        'San Francisco': 1.3, 'Phoenix': 0.9, 'Austin': 1.0,
        'Los Angeles': 1.2, 'Santa Monica': 1.1,
        # 默认
        'DEFAULT': 1.0,
    }

    def price(self, operator: str, pi_distribution: Dict[str, int],
              primary_city: str = 'DEFAULT') -> PricingResult:
        """
        计算保费

        Args:
            operator: 运营商名称
            pi_distribution: PI分布 {'PI-A': 10, 'PI-B': 5, 'PI-C': 2}
            primary_city: 主要运营城市

        Returns:
            定价结果
        """
        # 计算加权PI系数
        total_incidents = sum(pi_distribution.values())
        if total_incidents == 0:
            pi_coefficient = 1.0
        else:
            pi_coefficient = sum(
                self.PI_COEFFICIENTS.get(pi, 1.0) * count
                for pi, count in pi_distribution.items()
            ) / total_incidents

        # 运营商评级系数
        rating, operator_coeff = self._compute_operator_rating(pi_distribution)

        # 城市系数
        city_coeff = self.CITY_COEFFICIENTS.get(primary_city, 1.0)

        # 最终保费
        final_premium = (self.BASE_PREMIUM * pi_coefficient *
                        operator_coeff * city_coeff)

        return PricingResult(
            operator=operator,
            base_premium_per_vehicle_monthly=self.BASE_PREMIUM,
            pi_coefficient=pi_coefficient,
            operator_coefficient=operator_coeff,
            city_coefficient=city_coeff,
            final_premium=final_premium,
            operator_rating=rating,
            breakdown={
                'base': self.BASE_PREMIUM,
                'pi_adjustment': self.BASE_PREMIUM * (pi_coefficient - 1),
                'operator_adjustment': self.BASE_PREMIUM * pi_coefficient * (operator_coeff - 1),
                'city_adjustment': self.BASE_PREMIUM * pi_coefficient * operator_coeff * (city_coeff - 1),
            }
        )

    def _compute_operator_rating(self, pi_distribution: Dict) -> tuple:
        """
        计算运营商评级
        Returns: (评级字符串, 评级系数)
        """
        total = sum(pi_distribution.values())
        if total == 0:
            return ('★★★★★', 1.0)

        pi_c_pct = pi_distribution.get('PI-C', 0) / total * 100

        if pi_c_pct < 5:
            return ('★★★★★', 0.9)  # 5星，优惠10%
        elif pi_c_pct < 10:
            return ('★★★★', 1.0)   # 4星，标准
        elif pi_c_pct < 20:
            return ('★★★', 1.2)    # 3星，加成20%
        elif pi_c_pct < 40:
            return ('★★', 1.5)     # 2星，加成50%
        else:
            return ('★', 2.0)      # 1星，加成100%


if __name__ == '__main__':
    model = RiskPricingModel()

    # 示例：某运营商的PI分布
    result = model.price(
        operator='Operator X',
        pi_distribution={'PI-A': 15, 'PI-B': 20, 'PI-C': 10},
        primary_city='北京'
    )

    print(f"运营商: {result.operator}")
    print(f"评级: {result.operator_rating}")
    print(f"PI系数: {result.pi_coefficient:.2f}")
    print(f"城市系数: {result.city_coefficient:.2f}")
    print(f"月保费: ¥{result.final_premium:.0f}/车")
    print("\n明细:")
    for key, value in result.breakdown.items():
        print(f"  {key}: ¥{value:.0f}")
```

```python
#!/usr/bin/env python3
"""
FastAPI REST接口
提供实时风险评分和批量查询能力
"""
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from typing import List, Dict, Optional

app = FastAPI(
    title="ROAM Risk Pricing API",
    description="Robotaxi风险定价系统REST API",
    version="1.0.0"
)


class IncidentRequest(BaseModel):
    """单事件评分请求"""
    incident_id: str
    operator: str
    scenario_code: str
    speed_kmh: Optional[float] = None
    ttc_s: Optional[float] = None


class PricingRequest(BaseModel):
    """定价请求"""
    operator: str
    pi_distribution: Dict[str, int]
    primary_city: str = "DEFAULT"


class PricingResponse(BaseModel):
    """定价响应"""
    operator: str
    rating: str
    monthly_premium_cny: float
    pi_coefficient: float
    operator_coefficient: float
    city_coefficient: float


@app.get("/")
def root():
    return {
        "service": "ROAM Risk Pricing API",
        "version": "1.0.0",
        "endpoints": ["/price", "/pi", "/exposure", "/report"]
    }


@app.post("/price", response_model=PricingResponse)
def calculate_price(request: PricingRequest):
    """计算保费"""
    from risk_pricing_model import RiskPricingModel
    model = RiskPricingModel()
    result = model.price(
        operator=request.operator,
        pi_distribution=request.pi_distribution,
        primary_city=request.primary_city
    )
    return PricingResponse(
        operator=result.operator,
        rating=result.operator_rating,
        monthly_premium_cny=round(result.final_premium),
        pi_coefficient=result.pi_coefficient,
        operator_coefficient=result.operator_coefficient,
        city_coefficient=result.city_coefficient
    )


@app.post("/pi")
def calculate_pi(request: IncidentRequest):
    """计算单事件PI等级"""
    from pi_calculator import PICalculator
    # 简化示例
    calculator = PICalculator()
    return {
        "incident_id": request.incident_id,
        "pi_level": "PI-B",  # 实际需要完整的事件向量+基线数据
        "confidence": 0.7,
        "reasoning": "Behavior at P15 of baseline distribution"
    }


@app.get("/gb-thresholds")
def get_gb_thresholds():
    """获取GB标准残余风险阈值"""
    return {
        "collision_per_hour": 1e-4,
        "minor_injury_per_hour": 1e-5,
        "serious_injury_per_hour": 1e-6,
        "fatal_per_hour": 1e-7,
        "source": "中国GB《智能网联汽车 自动驾驶系统安全要求》附录D"
    }


if __name__ == '__main__':
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

```python
#!/usr/bin/env python3
"""
风险定价可视化仪表板
基于matplotlib/plotly生成图表
"""
import matplotlib.pyplot as plt
import numpy as np
from typing import Dict, List


class RiskDashboard:
    """风险可视化仪表板"""

    def plot_pi_distribution(self, operator: str, pi_counts: Dict[str, int],
                              save_path: str = None):
        """绘制PI分布饼图"""
        labels = []
        sizes = []
        colors = []

        color_map = {'PI-A': '#2d7a4f', 'PI-B': '#a67c2e', 'PI-C': '#b83a3a'}

        for pi_level in ['PI-A', 'PI-B', 'PI-C']:
            count = pi_counts.get(pi_level, 0)
            if count > 0:
                labels.append(f'{pi_level} ({count})')
                sizes.append(count)
                colors.append(color_map[pi_level])

        fig, ax = plt.subplots(figsize=(8, 8))
        ax.pie(sizes, labels=labels, colors=colors, autopct='%1.1f%%',
               startangle=90)
        ax.set_title(f'{operator} PI Distribution', fontsize=14, fontweight='bold')

        if save_path:
            plt.savefig(save_path, dpi=150, bbox_inches='tight')
        else:
            plt.show()

    def plot_rate_vs_gb(self, operator: str, rates: Dict[str, float],
                        gb_thresholds: Dict[str, float], save_path: str = None):
        """绘制事故率vs GB阈值对比柱状图（log scale）"""
        categories = list(rates.keys())
        actual = [rates[c] for c in categories]
        threshold = [gb_thresholds[c] for c in categories]

        x = np.arange(len(categories))
        width = 0.35

        fig, ax = plt.subplots(figsize=(10, 6))
        ax.bar(x - width/2, actual, width, label='Actual Rate', color='#c85a3a')
        ax.bar(x + width/2, threshold, width, label='GB Threshold', color='#2d7a4f')
        ax.set_yscale('log')
        ax.set_ylabel('Rate per Hour (log scale)')
        ax.set_title(f'{operator}: Rate vs GB Standard')
        ax.set_xticks(x)
        ax.set_xticklabels(categories, rotation=45)
        ax.legend()
        ax.grid(True, alpha=0.3, axis='y')

        if save_path:
            plt.savefig(save_path, dpi=150, bbox_inches='tight')
        else:
            plt.show()

    def plot_operator_comparison(self, operators_data: Dict[str, Dict],
                                  save_path: str = None):
        """绘制多运营商横向对比"""
        operators = list(operators_data.keys())
        pi_c_pcts = [operators_data[op].get('pi_c_pct', 0) for op in operators]

        fig, ax = plt.subplots(figsize=(10, 6))
        bars = ax.barh(operators, pi_c_pcts, color='#c85a3a')
        ax.set_xlabel('PI-C Percentage (%)')
        ax.set_title('Operator Comparison: PI-C Incidence Rate')

        for bar, pct in zip(bars, pi_c_pcts):
            ax.text(bar.get_width() + 0.5, bar.get_y() + bar.get_height()/2,
                   f'{pct:.1f}%', va='center')

        if save_path:
            plt.savefig(save_path, dpi=150, bbox_inches='tight')
        else:
            plt.show()


if __name__ == '__main__':
    dashboard = RiskDashboard()
    # 测试数据
    dashboard.plot_pi_distribution(
        operator='Test Operator',
        pi_counts={'PI-A': 15, 'PI-B': 20, 'PI-C': 10}
    )
```
