```python
#!/usr/bin/env python3
"""
ROAM事件数据解析器
读取YAML事件文件，提取精算所需字段
"""
import yaml
import re
from pathlib import Path
from dataclasses import dataclass, asdict
from typing import Optional, List, Dict
from datetime import datetime


@dataclass
class IncidentVector:
    """事件向量化表示（用于风险精算）"""
    id: str
    date: str
    operator: str
    scenario_code: str
    severity_level: int  # 0-4对应S0-S4
    urgency_level: int   # 0-3对应U0-U3
    city: str
    country: str
    road_type: str
    vehicles_affected: int
    duration_minutes: int
    injuries: int
    fatalities: int
    speed_estimate_kmh: Optional[float]
    ttc_estimate_s: Optional[float]
    outcome_score: float  # 综合伤亡+中断评分

    def to_dict(self) -> Dict:
        return asdict(self)


class EventParser:
    """事件数据解析器"""

    SEVERITY_MAP = {'S0': 0, 'S1': 1, 'S2': 2, 'S3': 3, 'S4': 4}
    URGENCY_MAP = {'U0': 0, 'U1': 1, 'U2': 2, 'U3': 3}

    def __init__(self, incidents_dir: str):
        self.incidents_dir = Path(incidents_dir)

    def parse_all(self) -> List[IncidentVector]:
        """解析所有事件"""
        vectors = []
        for f in sorted(self.incidents_dir.glob('*/ROAM-*.yaml')):
            try:
                vec = self.parse_file(f)
                if vec:
                    vectors.append(vec)
            except Exception as e:
                print(f"Parse error {f}: {e}")
        return vectors

    def parse_file(self, yaml_path: Path) -> Optional[IncidentVector]:
        """解析单个事件文件"""
        with open(yaml_path, 'r', encoding='utf-8') as f:
            data = yaml.safe_load(f)

        # 提取基础字段
        loc = data.get('location', {})
        impact = data.get('impact', {})

        # 从描述中提取速度和TTC（简化版NLP）
        desc = data.get('description', '')
        speed = self._extract_speed(desc)
        ttc = self._extract_ttc(desc)

        # 计算综合后果评分
        outcome_score = self._compute_outcome_score(
            injuries=impact.get('injuries', 0),
            fatalities=impact.get('fatalities', 0),
            traffic_disruption=impact.get('traffic_disruption', 'none'),
            duration=impact.get('duration_minutes', 0)
        )

        return IncidentVector(
            id=data.get('id', ''),
            date=data.get('date', ''),
            operator=data.get('operator', ''),
            scenario_code=data.get('scenario', {}).get('primary', ''),
            severity_level=self.SEVERITY_MAP.get(data.get('severity', 'S0'), 0),
            urgency_level=self.URGENCY_MAP.get(data.get('urgency', 'U0'), 0),
            city=loc.get('city', ''),
            country=loc.get('country', ''),
            road_type=loc.get('road_type', 'other'),
            vehicles_affected=impact.get('vehicles_affected', 1),
            duration_minutes=impact.get('duration_minutes', 0),
            injuries=impact.get('injuries', 0),
            fatalities=impact.get('fatalities', 0),
            speed_estimate_kmh=speed,
            ttc_estimate_s=ttc,
            outcome_score=outcome_score
        )

    def _extract_speed(self, text: str) -> Optional[float]:
        """从描述中提取速度估算"""
        # 匹配 "7 mph", "30 km/h", "60公里/小时" 等
        patterns = [
            (r'(\d+\.?\d*)\s*mph', 1.60934),  # mph to km/h
            (r'(\d+\.?\d*)\s*km/?h', 1.0),
            (r'(\d+\.?\d*)\s*公里/小时', 1.0),
        ]
        for pattern, multiplier in patterns:
            match = re.search(pattern, text, re.IGNORECASE)
            if match:
                return float(match.group(1)) * multiplier
        return None

    def _extract_ttc(self, text: str) -> Optional[float]:
        """从描述中提取TTC估算"""
        match = re.search(r'TTC[:\s]+(\d+\.?\d*)\s*s', text, re.IGNORECASE)
        return float(match.group(1)) if match else None

    def _compute_outcome_score(self, injuries: int, fatalities: int,
                                traffic_disruption: str, duration: int) -> float:
        """计算综合后果评分(0-100)"""
        score = 0.0
        # 伤亡权重
        score += fatalities * 40
        score += injuries * 10
        # 交通中断权重
        disruption_scores = {'none': 0, 'minor': 2, 'moderate': 5,
                             'severe': 15, 'critical': 30}
        score += disruption_scores.get(traffic_disruption, 0)
        # 持续时间权重
        score += min(duration / 10, 15)  # 每10分钟1分，上限15
        return min(score, 100.0)


if __name__ == '__main__':
    parser = EventParser('../incidents')
    vectors = parser.parse_all()
    print(f"Parsed {len(vectors)} incidents")
    for v in vectors[:3]:
        print(f"\n{v.id}: {v.operator} | {v.scenario_code} | "
              f"Sev={v.severity_level} | Score={v.outcome_score:.1f}")
```

```python
#!/usr/bin/env python3
"""
自然驾驶基线数据匹配器
对接DRIVEResearch航测NDD数据集
"""
import numpy as np
import pandas as pd
from dataclasses import dataclass
from typing import Dict, List, Optional, Tuple


@dataclass
class BaselineDistribution:
    """基线分布结果"""
    scenario_code: str
    road_type: str
    speed_range: str
    sample_size: int
    ttc_percentiles: Dict[str, float]  # P5, P25, P50, P75, P95
    thw_percentiles: Dict[str, float]
    acceleration_percentiles: Dict[str, float]
    lateral_accel_percentiles: Dict[str, float]
    confidence: float  # 0-1


class NDDMatcher:
    """自然驾驶数据基线匹配"""

    def __init__(self, ndd_data_path: str):
        """
        初始化
        ndd_data_path: DRIVEResearch航测数据库路径
        """
        self.ndd_path = ndd_data_path
        self._trajectories_cache: Dict[str, pd.DataFrame] = {}

    def match_baseline(self, scenario_code: str, road_type: str,
                       speed_range: str) -> Optional[BaselineDistribution]:
        """
        根据场景+道路+速度区间匹配基线分布

        Args:
            scenario_code: ROAM场景编码（如B1, C3）
            road_type: 道路类型
            speed_range: 速度区间（如"40-60"）

        Returns:
            基线分布对象
        """
        # 从NDD数据库加载匹配的轨迹
        trajectories = self._load_matching_trajectories(
            scenario_code, road_type, speed_range
        )

        if trajectories is None or len(trajectories) == 0:
            return None

        # 计算各参数的百分位分布
        return BaselineDistribution(
            scenario_code=scenario_code,
            road_type=road_type,
            speed_range=speed_range,
            sample_size=len(trajectories),
            ttc_percentiles=self._compute_percentiles(trajectories['ttc']),
            thw_percentiles=self._compute_percentiles(trajectories['thw']),
            acceleration_percentiles=self._compute_percentiles(trajectories['accel']),
            lateral_accel_percentiles=self._compute_percentiles(trajectories['lat_accel']),
            confidence=self._compute_confidence(len(trajectories))
        )

    def _load_matching_trajectories(self, scenario_code: str,
                                     road_type: str,
                                     speed_range: str) -> Optional[pd.DataFrame]:
        """从NDD数据库加载匹配的轨迹"""
        cache_key = f"{scenario_code}_{road_type}_{speed_range}"
        if cache_key in self._trajectories_cache:
            return self._trajectories_cache[cache_key]

        # 实际实现中连接到DRIVEResearch数据库
        # 此处模拟返回一个样本数据集
        # 真实实现: df = query_ndd_database(scenario_code, road_type, speed_range)

        # 模拟数据
        n_samples = self._estimate_sample_size(scenario_code, road_type, speed_range)
        if n_samples == 0:
            return None

        np.random.seed(hash(cache_key) % 2**32)
        df = pd.DataFrame({
            'ttc': np.random.lognormal(mean=1.8, sigma=0.5, size=n_samples),
            'thw': np.random.lognormal(mean=0.5, sigma=0.4, size=n_samples),
            'accel': np.random.normal(loc=0, scale=1.2, size=n_samples),
            'lat_accel': np.random.normal(loc=0, scale=0.8, size=n_samples),
        })
        self._trajectories_cache[cache_key] = df
        return df

    def _estimate_sample_size(self, scenario_code: str, road_type: str,
                               speed_range: str) -> int:
        """估算匹配场景的样本量"""
        # 常见场景样本量较大
        common_scenarios = {'B1', 'C1', 'E2', 'F2'}
        if scenario_code in common_scenarios:
            return 5000
        elif scenario_code.startswith(('A', 'D')):
            return 50  # 罕见场景
        else:
            return 500

    @staticmethod
    def _compute_percentiles(series: pd.Series) -> Dict[str, float]:
        """计算P5/P25/P50/P75/P95百分位"""
        return {
            'P5': float(np.percentile(series, 5)),
            'P25': float(np.percentile(series, 25)),
            'P50': float(np.percentile(series, 50)),
            'P75': float(np.percentile(series, 75)),
            'P95': float(np.percentile(series, 95)),
        }

    @staticmethod
    def _compute_confidence(sample_size: int) -> float:
        """基于样本量计算置信度"""
        if sample_size >= 1000:
            return 0.9
        elif sample_size >= 100:
            return 0.7
        elif sample_size >= 30:
            return 0.5
        else:
            return 0.3


def speed_to_range(speed_kmh: float) -> str:
    """速度转换到区间"""
    if speed_kmh < 20:
        return "0-20"
    elif speed_kmh < 40:
        return "20-40"
    elif speed_kmh < 60:
        return "40-60"
    elif speed_kmh < 80:
        return "60-80"
    elif speed_kmh < 100:
        return "80-100"
    else:
        return "100-120"


if __name__ == '__main__':
    matcher = NDDMatcher('/path/to/driveresearch/ndd')
    baseline = matcher.match_baseline(
        scenario_code='B1',
        road_type='intersection',
        speed_range='20-40'
    )
    if baseline:
        print(f"Sample size: {baseline.sample_size}")
        print(f"Confidence: {baseline.confidence}")
        print(f"TTC P5: {baseline.ttc_percentiles['P5']:.2f}s")
        print(f"TTC P25: {baseline.ttc_percentiles['P25']:.2f}s")
        print(f"TTC P50: {baseline.ttc_percentiles['P50']:.2f}s")
```

## 模块3：可预防性指数计算

### 3.1 pi_calculator.py

```python
#!/usr/bin/env python3
"""
可预防性指数（Preventability Index, PI）计算模块
"""
import numpy as np
from dataclasses import dataclass
from enum import Enum
from typing import Dict, Optional


class PILevel(Enum):
    """PI等级枚举"""
    PI_A = "PI-A"  # 不可预防
    PI_B = "PI-B"  # 边界可预防
    PI_C = "PI-C"  # 应可预防
    UNKNOWN = "Unknown"


@dataclass
class PIResult:
    """PI计算结果"""
    incident_id: str
    pi_level: PILevel
    confidence: float
    behavior_percentile: Optional[float]  # 事故行为在基线分布中的百分位
    baseline_sample_size: int
    reasoning: str


class PICalculator:
    """可预防性指数计算器"""

    # PI等级判定阈值（基于基线百分位）
    PI_A_THRESHOLD = 5   # 事故行为<P5 → 不可预防
    PI_B_THRESHOLD = 25  # 事故行为P5-P25 → 边界可预防
    # >P25 → PI-C 应可预防

    def calculate(self, incident_vector, baseline_distribution) -> PIResult:
        """
        计算单起事件的PI等级

        Args:
            incident_vector: 事件向量（含行为参数）
            baseline_distribution: 基线分布

        Returns:
            PI计算结果
        """
        # 如果没有基线数据，返回UNKNOWN
        if baseline_distribution is None:
            return PIResult(
                incident_id=incident_vector.id,
                pi_level=PILevel.UNKNOWN,
                confidence=0.0,
                behavior_percentile=None,
                baseline_sample_size=0,
                reasoning="No baseline data available for scenario"
            )

        # 定位事故行为在基线分布中的百分位
        percentile = self._locate_percentile(incident_vector, baseline_distribution)

        if percentile is None:
            return PIResult(
                incident_id=incident_vector.id,
                pi_level=PILevel.UNKNOWN,
                confidence=baseline_distribution.confidence * 0.5,
                behavior_percentile=None,
                baseline_sample_size=baseline_distribution.sample_size,
                reasoning="Cannot extract behavior parameters from incident"
            )

        # 判定PI等级
        if percentile < self.PI_A_THRESHOLD:
            pi_level = PILevel.PI_A
            reasoning = (f"事故行为位于基线P{percentile:.0f}，属于极端不利情形，"
                        f"人类驾驶员也难以避免（不可预防）")
        elif percentile < self.PI_B_THRESHOLD:
            pi_level = PILevel.PI_B
            reasoning = (f"事故行为位于基线P{percentile:.0f}，少数优秀驾驶员可避免"
                        f"（边界可预防）")
        else:
            pi_level = PILevel.PI_C
            reasoning = (f"事故行为位于基线P{percentile:.0f}，大多数驾驶员可避免，"
                        f"属于系统缺陷（应可预防）")

        return PIResult(
            incident_id=incident_vector.id,
            pi_level=pi_level,
            confidence=baseline_distribution.confidence,
            behavior_percentile=percentile,
            baseline_sample_size=baseline_distribution.sample_size,
            reasoning=reasoning
        )

    def _locate_percentile(self, incident_vector, baseline) -> Optional[float]:
        """
        定位事故行为在基线分布中的百分位位置

        使用TTC作为主要参数（如果可获得），否则使用加速度等其他参数
        """
        # 优先使用TTC
        if incident_vector.ttc_estimate_s is not None:
            ttc_value = incident_vector.ttc_estimate_s
            percentiles = baseline.ttc_percentiles
            return self._interpolate_percentile(ttc_value, percentiles)

        # 回退到使用speed（通过加速度间接反映）
        if incident_vector.speed_estimate_kmh is not None:
            # 如果速度超过基线P95，判定为P5以下（罕见高速）
            # 简化实现：使用outcome_score反推
            return None

        return None

    @staticmethod
    def _interpolate_percentile(value: float, percentiles: Dict[str, float]) -> float:
        """根据百分位分布表，线性插值定位具体百分位"""
        p5 = percentiles['P5']
        p25 = percentiles['P25']
        p50 = percentiles['P50']
        p75 = percentiles['P75']
        p95 = percentiles['P95']

        # 注意：对于TTC，小值表示危险（低于P5更危险）
        # 对于THW, acceleration, 同理
        if value <= p5:
            return 5 * (value / p5) if p5 > 0 else 0
        elif value <= p25:
            return 5 + 20 * (value - p5) / (p25 - p5)
        elif value <= p50:
            return 25 + 25 * (value - p25) / (p50 - p25)
        elif value <= p75:
            return 50 + 25 * (value - p50) / (p75 - p50)
        elif value <= p95:
            return 75 + 20 * (value - p75) / (p95 - p75)
        else:
            return min(95 + 5 * (value - p95) / p95, 100)


def compute_pi_summary(pi_results: list) -> Dict:
    """统计PI分布摘要"""
    from collections import Counter
    counter = Counter(r.pi_level for r in pi_results)
    total = len(pi_results)
    return {
        'total': total,
        'pi_a_count': counter[PILevel.PI_A],
        'pi_b_count': counter[PILevel.PI_B],
        'pi_c_count': counter[PILevel.PI_C],
        'unknown_count': counter[PILevel.UNKNOWN],
        'pi_a_pct': counter[PILevel.PI_A] / total * 100 if total > 0 else 0,
        'pi_b_pct': counter[PILevel.PI_B] / total * 100 if total > 0 else 0,
        'pi_c_pct': counter[PILevel.PI_C] / total * 100 if total > 0 else 0,
    }


if __name__ == '__main__':
    calculator = PICalculator()
    print("PI Calculator initialized")
    print(f"PI-A threshold: <P{calculator.PI_A_THRESHOLD}")
    print(f"PI-B threshold: P{calculator.PI_A_THRESHOLD}-P{calculator.PI_B_THRESHOLD}")
    print(f"PI-C threshold: >P{calculator.PI_B_THRESHOLD}")
```
