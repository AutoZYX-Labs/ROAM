## 第5部分：参考架构映射表处理

### 5.1 scenario_to_layer_mapping.py

```python
#!/usr/bin/env python3
"""场景-处置映射表处理工具"""
from typing import Dict, List, Tuple
from enum import Enum

class Layer(Enum):
    L1 = "Layer 1: AI Autonomous Response"
    L2 = "Layer 2: AI-Assisted + Human Confirm"
    L3 = "Layer 3: Remote Driving / On-Site"

# 场景-处置层级映射表
SCENARIO_LAYER_MAP: Dict[str, Tuple[Layer, str]] = {
    'A1': (Layer.L1, "Trigger pre-programmed MRC; escalate to L3 for fleet recovery"),
    'A2': (Layer.L3, "Rollback OTA update, human decision required"),
    'A3': (Layer.L1, "Edge-cached decision; L3 for power restoration"),
    'A4': (Layer.L1, "Edge fallback, local autonomy continues"),
    'B1': (Layer.L2, "AI pre-computes options, human confirms intersection action"),
    'B2': (Layer.L2, "Human disambiguates object type"),
    'B3': (Layer.L1, "ODD exit, safe stop"),
    'B4': (Layer.L1, "Conservative response, defensive driving"),
    'B5': (Layer.L2, "Human assists with localization"),
    'C1': (Layer.L1, "Auto-restart; L2 if repeats"),
    'C2': (Layer.L2, "Review braking parameters"),
    'C3': (Layer.L2, "Human intervention for lane changes"),
    'C4': (Layer.L3, "Damage assessment dispatch"),
    'C5': (Layer.L3, "Incident response, emergency services"),
    'D1': (Layer.L1, "Redundant sensor fallback; L2 if degraded"),
    'D2': (Layer.L3, "Tow dispatch"),
    'D3': (Layer.L3, "Emergency response, passenger evacuation"),
    'D4': (Layer.L3, "Immediate safe-stop + rescue"),
    'E1': (Layer.L3, "Incident scene response"),
    'E2': (Layer.L2, "Reroute planning"),
    'E3': (Layer.L2, "Emergency vehicle priority detection"),
    'E4': (Layer.L2, "Human interprets police gestures"),
    'F1': (Layer.L1, "Unlock doors, voice guidance; L3 if critical"),
    'F2': (Layer.L2, "Alternate dropoff location"),
    'F3': (Layer.L2, "Voice intervention"),
    'F4': (Layer.L3, "Ambulance dispatch"),
}

def get_layer_for_scenario(scenario_code: str) -> Tuple[Layer, str]:
    """获取场景的推荐处置层级"""
    return SCENARIO_LAYER_MAP.get(scenario_code.upper(),
                                   (Layer.L2, "Default: human review"))

def get_scenarios_by_layer(layer: Layer) -> List[str]:
    """获取某层级下的所有场景"""
    return [code for code, (l, _) in SCENARIO_LAYER_MAP.items() if l == layer]

def compute_layer_distribution() -> Dict[str, float]:
    """计算场景在三层中的分布百分比"""
    from collections import Counter
    layers = [l for l, _ in SCENARIO_LAYER_MAP.values()]
    counts = Counter(layers)
    total = len(layers)
    return {l.name: counts[l]/total*100 for l in Layer}

if __name__ == '__main__':
    print("ROAM Scenario-to-Layer Mapping")
    print("="*60)
    dist = compute_layer_distribution()
    for layer_name, pct in dist.items():
        print(f"{layer_name}: {pct:.1f}% of scenarios")

    print("\nLayer 1 scenarios:", get_scenarios_by_layer(Layer.L1))
    print("Layer 2 scenarios:", get_scenarios_by_layer(Layer.L2))
    print("Layer 3 scenarios:", get_scenarios_by_layer(Layer.L3))
```

## 第6部分：KPI计算工具

### 6.1 kpi_calculator.py

```python
#!/usr/bin/env python3
"""ROAM KPI计算工具"""
import yaml
import glob
from pathlib import Path
from dataclasses import dataclass
from typing import List, Dict
from datetime import datetime

@dataclass
class KPI:
    """KPI定义"""
    name: str
    description: str
    target: str
    unit: str

KPIs = {
    'MTTR': KPI('Mean Time To Resolution', 'Avg time from anomaly to resolved',
                '<5min (AI), <15min (human)', 'minutes'),
    'AI_RESOLUTION_RATE': KPI('AI Autonomous Resolution Rate',
                               'Anomalies resolved by AI alone', '>70%', 'percent'),
    'FALSE_ESCALATION_RATE': KPI('False Escalation Rate',
                                  'L1-solvable but escalated to L2', '<10%', 'percent'),
    'MISSED_ESCALATION_RATE': KPI('Missed Escalation Rate',
                                   'Not escalated but should be', '<1%', 'percent'),
    'PASSENGER_TRAPPED_DURATION': KPI('Passenger Trapped Duration',
                                       'Time passenger unable to exit', '<10min', 'minutes'),
    'TRAFFIC_RECOVERY_TIME': KPI('Traffic Recovery Time',
                                  'Time from stall to road clearance', '<20min', 'minutes'),
    'OPERATOR_VEHICLE_RATIO': KPI('Operator-to-Vehicle Ratio',
                                   'Remote operators per vehicle', '1:80-1:100', 'ratio'),
    'SCENARIO_COVERAGE_RATE': KPI('Scenario Coverage Rate',
                                   'Known scenarios in database', '>90%', 'percent'),
}

def calculate_mttr(incidents: List[Dict]) -> float:
    """计算MTTR"""
    durations = [i.get('impact', {}).get('duration_minutes', 0)
                 for i in incidents
                 if i.get('impact', {}).get('duration_minutes')]
    return sum(durations)/len(durations) if durations else 0

def calculate_severity_distribution(incidents: List[Dict]) -> Dict[str, int]:
    """计算严重度分布"""
    dist = {f'S{i}': 0 for i in range(5)}
    for inc in incidents:
        s = inc.get('severity', 'S0')
        dist[s] = dist.get(s, 0) + 1
    return dist

def generate_kpi_report(incidents_dir='incidents') -> str:
    """生成KPI报告"""
    incidents = []
    for f in Path(incidents_dir).glob('*/ROAM-*.yaml'):
        with open(f, 'r', encoding='utf-8') as fp:
            incidents.append(yaml.safe_load(fp))

    report = []
    report.append("# ROAM KPI Report")
    report.append(f"\nGenerated: {datetime.now().strftime('%Y-%m-%d %H:%M')}")
    report.append(f"Total incidents: {len(incidents)}")

    mttr = calculate_mttr(incidents)
    report.append(f"\n## MTTR: {mttr:.1f} minutes (target: <5 AI, <15 human)")

    dist = calculate_severity_distribution(incidents)
    report.append(f"\n## Severity Distribution:")
    for sev, count in dist.items():
        pct = count/len(incidents)*100 if incidents else 0
        report.append(f"  {sev}: {count} ({pct:.1f}%)")

    return '\n'.join(report)

if __name__ == '__main__':
    print(generate_kpi_report())
```
