## 第2部分：事件数据验证工具

### 2.1 validate_incidents.py

```python
#!/usr/bin/env python3
"""
ROAM事件数据库验证工具
验证所有YAML事件文件是否符合Schema约束
"""
import json
import glob
import yaml
import sys
from pathlib import Path
from jsonschema import validate, ValidationError

def load_schema(schema_path):
    """加载JSON Schema"""
    with open(schema_path, 'r', encoding='utf-8') as f:
        return json.load(f)

def validate_incident(incident_data, schema):
    """验证单个事件文件"""
    try:
        validate(instance=incident_data, schema=schema)
        return True, None
    except ValidationError as e:
        return False, str(e)

def check_id_uniqueness(incidents):
    """检查事件ID唯一性"""
    ids = [inc['id'] for inc in incidents]
    duplicates = [x for x in ids if ids.count(x) > 1]
    return len(duplicates) == 0, set(duplicates)

def check_sources_accessible(incidents):
    """检查来源URL可访问性（需联网）"""
    import requests
    broken_urls = []
    for inc in incidents:
        for src in inc.get('sources', []):
            url = src.get('url', '')
            try:
                r = requests.head(url, timeout=10, allow_redirects=True)
                if r.status_code >= 400:
                    broken_urls.append((inc['id'], url, r.status_code))
            except Exception as e:
                broken_urls.append((inc['id'], url, str(e)))
    return broken_urls

def main():
    schema_path = Path(__file__).parent.parent / 'incidents/schema.json'
    schema = load_schema(schema_path)

    incidents_dir = Path(__file__).parent.parent / 'incidents'
    yaml_files = list(incidents_dir.glob('*/ROAM-*.yaml'))

    print(f"ROAM Incident Database Validation")
    print(f"{'='*50}")
    print(f"Schema: {schema_path}")
    print(f"Found {len(yaml_files)} incident files\n")

    incidents = []
    errors = []

    for f in yaml_files:
        with open(f, 'r', encoding='utf-8') as fp:
            try:
                data = yaml.safe_load(fp)
                incidents.append(data)
            except yaml.YAMLError as e:
                errors.append((f.name, f"YAML parse error: {e}"))
                continue

        ok, err = validate_incident(data, schema)
        if not ok:
            errors.append((f.name, err))

    # ID uniqueness check
    ok_ids, duplicates = check_id_uniqueness(incidents)
    if not ok_ids:
        print(f"❌ Duplicate IDs found: {duplicates}")
    else:
        print(f"✓ All {len(incidents)} event IDs unique")

    # Report errors
    if errors:
        print(f"\n❌ Validation errors in {len(errors)} files:")
        for fname, err in errors:
            print(f"  - {fname}: {err[:100]}")
        sys.exit(1)
    else:
        print(f"✓ All {len(incidents)} incidents pass schema validation")

    # Stats
    print(f"\nStatistics:")
    print(f"  Operators: {len(set(i['operator'] for i in incidents))}")
    print(f"  Countries: {len(set(i['location']['country'] for i in incidents if 'country' in i.get('location', {})))}")
    severities = {}
    for i in incidents:
        s = i.get('severity', 'Unknown')
        severities[s] = severities.get(s, 0) + 1
    print(f"  Severity distribution: {severities}")

if __name__ == '__main__':
    main()
```

### 2.2 query_incidents.py

```python
#!/usr/bin/env python3
"""ROAM事件数据库查询工具"""
import yaml
import glob
import argparse
from pathlib import Path

def load_all_incidents(base_dir='incidents'):
    """加载所有事件数据"""
    incidents = []
    for f in sorted(Path(base_dir).glob('*/ROAM-*.yaml')):
        with open(f, 'r', encoding='utf-8') as fp:
            data = yaml.safe_load(fp)
            data['_file'] = str(f)
            incidents.append(data)
    return incidents

def filter_by_operator(incidents, operator):
    """按运营商筛选"""
    return [i for i in incidents if operator.lower() in i['operator'].lower()]

def filter_by_severity(incidents, min_level='S0', max_level='S4'):
    """按严重度范围筛选"""
    levels = ['S0', 'S1', 'S2', 'S3', 'S4']
    min_idx = levels.index(min_level)
    max_idx = levels.index(max_level)
    return [i for i in incidents
            if min_idx <= levels.index(i.get('severity', 'S0')) <= max_idx]

def filter_by_scenario(incidents, scenario_code):
    """按场景编码筛选"""
    return [i for i in incidents
            if i.get('scenario', {}).get('primary', '').startswith(scenario_code)]

def filter_by_date_range(incidents, start_date, end_date):
    """按日期范围筛选"""
    return [i for i in incidents
            if start_date <= i.get('date', '') <= end_date]

def print_incident_summary(incident):
    """打印事件摘要"""
    print(f"\n[{incident['id']}] {incident['date']} - {incident['operator']}")
    print(f"  Location: {incident['location'].get('city', 'N/A')}, "
          f"{incident['location'].get('country', 'N/A')}")
    print(f"  Scenario: {incident['scenario']['primary']} | "
          f"Severity: {incident['severity']} | "
          f"Urgency: {incident.get('urgency', 'N/A')}")
    desc = incident.get('description', '').strip()
    print(f"  {desc[:150]}{'...' if len(desc) > 150 else ''}")

def main():
    parser = argparse.ArgumentParser(description='ROAM Incident Query Tool')
    parser.add_argument('--operator', help='Filter by operator name')
    parser.add_argument('--severity-min', default='S0', help='Min severity (S0-S4)')
    parser.add_argument('--severity-max', default='S4', help='Max severity (S0-S4)')
    parser.add_argument('--scenario', help='Filter by scenario code (A/B/C/D/E/F or A1/B2/etc.)')
    parser.add_argument('--from-date', help='Start date (YYYY-MM-DD)')
    parser.add_argument('--to-date', help='End date (YYYY-MM-DD)')
    parser.add_argument('--base-dir', default='incidents', help='Incidents directory')
    args = parser.parse_args()

    incidents = load_all_incidents(args.base_dir)
    print(f"Loaded {len(incidents)} total incidents")

    if args.operator:
        incidents = filter_by_operator(incidents, args.operator)
    if args.scenario:
        incidents = filter_by_scenario(incidents, args.scenario)
    if args.from_date and args.to_date:
        incidents = filter_by_date_range(incidents, args.from_date, args.to_date)
    incidents = filter_by_severity(incidents, args.severity_min, args.severity_max)

    print(f"Query results: {len(incidents)} incidents match filters")
    for inc in incidents:
        print_incident_summary(inc)

if __name__ == '__main__':
    main()
```
